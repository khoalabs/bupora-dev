	🔥 Technical Launch SEO Checklist chuẩn cho Flask app
		🎯 Mục tiêu:
			👉 Index nhanh
			👉 Crawl sạch
			👉 Không lỗi technical ngu
			👉 Chuẩn để scale lâu dài

			🧱 Infrastructure
			🔍 Crawl & Index
			🧭 On-page Technical
			⚡ Performance
			📊 Tracking
		🧱 I. Infrastructure (Bắt buộc trước khi mở index)
			1. Production Setup
				✔ App factory pattern (bạn đã chọn đúng hướng)
				✔ DEBUG = False
				✔ Proper error pages (404, 500 custom)
				✔ HTTPS (Let’s Encrypt)
				✔ www → non-www redirect (chọn 1)
				✔ HTTP → HTTPS redirect
				Không được để:
					http:// và https:// cùng tồn tại
					example.com và www.example.com → index song song
			2. URL Structure Chuẩn SEO
				Phải là:
					/breeds/golden-retriever/
					/care/feeding/how-much-to-feed-golden/
					/shop/golden-retriever/
				Không:
					/post?id=123
					/breed?name=golden
				✔ Flask route phải clean, không query rác.
		🔍 II. Crawl & Index
			3. robots.txt
				Tạo route:
					/robots.txt
				Ví dụ:
					User-agent: *
					Allow: /
					Sitemap: https://bupora.com/sitemap.xml

					Disallow: /

					/sitemap.xml

			5. Canonical Tag
				Mỗi page cần:
					<link rel="canonical" href="https://bupora.com/current-url/" />
				Giúp tránh duplicate:
					Trailing slash vs no slash
					UTM parameters

	🗺️ sitemap generator chuẩn SEO cho Flask
		🚀 Mục tiêu
			Dynamic (không hardcode URL)
			Hỗ trợ bài blog, breed, shop landing
			Chuẩn XML
			Dễ mở rộng sau này
		🎯 Yêu cầu trước khi viết
			Giả sử bạn có model kiểu:
				Post (care articles)
				Breed
				ShopLanding
			Hoặc ít nhất có slug + updated_at.
			Nếu chưa có DB hoàn chỉnh, ta vẫn làm dạng template để sau này thay model vào.
		🧱 Bước 1: Tạo route /sitemap.xml
			# app/main/routes.py
			from flask import Blueprint, render_template, current_app
			from datetime import datetime

			main = Blueprint("main", __name__)

			@main.route("/sitemap.xml", methods=["GET"])
			def sitemap():
					pages = []

					# Static pages
					static_pages = [
							"main.index",
							"main.about",
							"breeds.index",
					]

					for endpoint in static_pages:
							url = current_app.url_for(endpoint, _external=True)
							pages.append({
									"loc": url,
									"lastmod": datetime.utcnow().date().isoformat(),
									"changefreq": "weekly",

					import os
					from datetime import datetime
					from app import create_app
					from app.models import Post, Breed, ShopLanding
					from flask import url_for

					BASE_URL = "https://bupora.com"

					app = create_app()

					def generate():

							with app.app_context():

									urls = []

									# Homepage
									urls.append({
											"loc": f"{BASE_URL}/",
											"lastmod": datetime.utcnow().date().isoformat(),
											"priority": "1.0"
									})

									# Breeds
									breeds = Breed.query.all()
									for breed in breeds:
											urls.append({
													"loc": f"{BASE_URL}/breeds/{breed.slug}/",
													"lastmod": breed.updated_at.date().isoformat(),
													"priority": "0.9"
											})

									# Care Articles
									posts = Post.query.filter_by(published=True).all()
									for post in posts:
											urls.append({
													"loc": f"{BASE_URL}/care/{post.category}/{post.slug}/",
													"lastmod": post.updated_at.date().isoformat(),
													"priority": "0.7"
											})

									# Shop Landing
									shops = ShopLanding.query.all()
									for shop in shops:
											urls.append({
													"loc": f"{BASE_URL}/shop/{shop.slug}/",
													"lastmod": shop.updated_at.date().isoformat(),
													"priority": "0.6"
											})

									# Build XML
									xml = ['<?xml version="1.0" encoding="UTF-8"?>']
									xml.append('<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">')

									for url in urls:
											xml.append("<url>")
											xml.append(f"<loc>{url['loc']}</loc>")
											xml.append(f"<lastmod>{url['lastmod']}</lastmod>")
											xml.append("<changefreq>monthly</changefreq>")
											xml.append(f"<priority>{url['priority']}</priority>")
											xml.append("</url>")

									xml.append("</urlset>")

									sitemap_path = os.path.join(app.root_path, "static", "sitemap.xml")

									with open(sitemap_path, "w", encoding="utf-8") as f:
											f.write("\n".join(xml))

									print("Sitemap generated successfully.")

					if __name__ == "__main__":
							generate()

				import requests
				import logging

				GOOGLE_PING_URL = "https://www.google.com/ping"

				def ping_google(sitemap_url):
						try:
								response = requests.get(
										GOOGLE_PING_URL,
										params={"sitemap": sitemap_url},
										timeout=10
								)

								if response.status_code == 200:
										logging.info("✅ Successfully pinged Google.")
								else:
										logging.warning(f"⚠ Google ping returned {response.status_code}")

						except Exception as e:
								logging.error(f"❌ Google ping failed: {e}")

								h = hashlib.md5()
								with open(path, "rb") as f:
										h.update(f.read())
								return h.hexdigest()

							SITEMAP_PATH = "static/sitemap.xml"
							TEMP_PATH = "static/sitemap_temp.xml"

							def main():
									generate_sitemap(output_path=TEMP_PATH)

									if not Path(SITEMAP_PATH).exists():
											Path(TEMP_PATH).rename(SITEMAP_PATH)
											ping_google(SITEMAP_PUBLIC_URL)
											return

									old_hash = file_hash(SITEMAP_PATH)
									new_hash = file_hash(TEMP_PATH)

									if old_hash != new_hash:
											Path(TEMP_PATH).replace(SITEMAP_PATH)
											ping_google(SITEMAP_PUBLIC_URL)
											print("Sitemap updated + Google pinged")
									else:
											Path(TEMP_PATH).unlink()
											print("No change, skip ping")
